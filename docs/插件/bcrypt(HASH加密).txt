使用：

const bcrypt = require('bcryptjs');

// 加密密碼
bcrypt.hash('myPassword', 10, (err, hash) => {
    if (err) throw err;
    console.log(hash); // 打印加密後的密碼
});

// 比對密碼
bcrypt.compare('myPassword', hash, (err, result) => {
    if (err) throw err;
    console.log(result); // 比對結果，true 或 false
});


優勢： 不需要原生 C++ 編譯，適合需要在不支持原生編譯的環境（例如某些無法安裝 C++ 編譯器的系統）中使用。
